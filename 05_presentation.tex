\begin{frame}{Research Question}
\phantomsection\label{research-question}
\textbf{Does energy efficiency affect rental prices?}

\begin{itemize}
\tightlist
\item
  \textbf{Motivation:} Increasing energy costs and climate regulations.
\item
  \textbf{Hypothesis:} Efficient apartments (A+, A, B) command a price
  premium (``Green Premium'').
\item
  \textbf{Method:} Hedonic regression on ImmoScout24 data.
\end{itemize}
\end{frame}

\begin{frame}{Descriptive Evidence}
\phantomsection\label{descriptive-evidence}
\begin{figure}[H]

{\centering \includegraphics[width=0.9\linewidth,height=\textheight,keepaspectratio]{output/figures/png/fig01_boxplot_raw.png}

}

\caption{Rent Distribution by Energy Class}

\end{figure}%

\begin{itemize}
\tightlist
\item
  \textbf{Observation:} Rent tends to be higher for efficient classes
  (A+/A).
\item
  \textbf{Caveat:} Raw data does not control for location or size.
\end{itemize}
\end{frame}

\begin{frame}{The Green Premium (Results)}
\phantomsection\label{the-green-premium-results}
\begin{figure}[H]

{\centering \includegraphics[width=0.9\linewidth,height=\textheight,keepaspectratio]{output/figures/png/fig02_green_premium.png}

}

\caption{Regression Coefficients}

\end{figure}%

\begin{itemize}
\tightlist
\item
  \textbf{Result:} Significant premium for Classes A+, A, and B relative
  to Class C.
\item
  \textbf{Discount:} Inefficient classes (F, G, H) trade at a discount.
\end{itemize}
\end{frame}

\begin{frame}{Urban vs.~Rural Markets}
\phantomsection\label{urban-vs.-rural-markets}
\begin{figure}[H]

{\centering \includegraphics[width=0.9\linewidth,height=\textheight,keepaspectratio]{output/figures/png/fig03_interaction_effect.png}

}

\caption{Interaction Effects}

\end{figure}%

\begin{itemize}
\tightlist
\item
  \textbf{Urban (Red):} Steep premium for efficiency.
\item
  \textbf{Rural (Green):} Flatter curve; efficiency matters less.
\item
  \textbf{Conclusion:} The ``Green Premium'' is stronger in cities.
\end{itemize}
\end{frame}

\begin{frame}{Conclusion}
\phantomsection\label{conclusion}
\begin{enumerate}
\tightlist
\item
  \textbf{Energy Efficiency Matters:} It is capitalized into rental
  prices.
\item
  \textbf{Market Heterogeneity:} The effect is driven largely by urban
  markets.
\item
  \textbf{Policy Implication:} Decarbonization incentives work
  differently in cities vs.~countryside.
\end{enumerate}

\textbf{Thank You!}
\end{frame}
